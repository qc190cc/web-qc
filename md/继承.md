
#继承
###原型链
 - 继承其他构造函数的原型，不会继承对象

		function Foo(){
		    this.age=20;
		    this.gender="male"
		}
		Foo.prototype.sayHello=function(){
		    console.log("hello")
		}
		function Foo2(){}
		Foo2.prototype=Object.create(Foo.prototype)
		Foo2.prototype.constructor=Foo2;
		let s=new Foo2()
		s.sayHello()
		 //输出为 hello，其实是一种创建对象的方式，
		和不写Objert.create区别不大，Foo2.prototype.constructor=Foo2也可以不写。

 - 对象冒充，不会继承原型

		function Foo(){
		    this.age=20;
		    this.gender="male"
		}
		Foo.prototype.sayHello=function(){
		    console.log("hello")
		}
		function Foo2(){
		    Foo.call(this)
		}
		let s=new Foo2()
		console.log(s.age) //输出为20
		
 - 混合模式，可以冒充对象加继承原型

		function Foo(){
		    this.age=20;
		    this.gender="male"
		}
		Foo.prototype.sayHello=function(){
		    console.log("hello")
		}
		function Foo2(){
		    Foo.call(this)
		}
		Foo2.prototype=Foo.prototype;
		let s=new Foo2()
		console.log(s.age)  //输出为 20
		s.sayHello()		//输出为 hello
